    val id: ExecId,
    val namespaceId: NamespaceId,
    val groupId: GroupId,
    val inputs: ExecInputs,
    val state: State,
    val code: CodeValue,
    val token: ExecToken = ExecToken("let_me_in"), // FIXME

    val events: List<Event> = listOf()
